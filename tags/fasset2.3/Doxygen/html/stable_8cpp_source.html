00753 void stable::controlled(double heatOP, double outsideAirTemp, double *supplementaryHeat)
00754 {
00755         double surfaceArea = planArea + wallArea; //sq meters - surface area of house
00756 
00757         double meanThermalTrans = (planArea * thermalTransRoof + wallArea * thermalTransWall)/surfaceArea;   //in W per metre square per K
00758         outsideAirTemp+=273.13;
00759         double deltaTemper=0.0;
00760         *supplementaryHeat=0.0;
00761         ventilationRate=0.0;
00762         double airDensity = GetdensityAir(GetstandardAirPressure(),targetTemperature,GetsaturatedWaterVapourPressure(targetTemperature));
00763         //should include a water balance!
00764         if (targetTemperature>outsideAirTemp) //it will be possible to maintain target temp
00765         {
00766                 ventilationRate=((heatOP/(targetTemperature-outsideAirTemp))- meanThermalTrans*surfaceArea)
00767                                                                                                                 /(GetspecificHeatCapAir()*airDensity);
00768                 insideTemperature=targetTemperature  - 273.13;
00769         }
00770         if ((ventilationRate > maxVentilation)||(targetTemperature<=outsideAirTemp)) //it will not be possible to maintain target temp, so set ventilation to max
00771         {
00772                 ventilationRate=maxVentilation;
00773                 deltaTemper=heatOP/(GetspecificHeatCapAir()*airDensity* ventilationRate + meanThermalTrans*surfaceArea);
00774                 //calculate the inside temperature
00775                 insideTemperature= outsideAirTemp + deltaTemper - 273.13;
00776         }
00777         if (ventilationRate< minVentilation)  //to keep the animals healthy, there needs to be some ventilation
00778         {
00779                 ventilationRate= minVentilation;
00780                 //calculate the supplementary heat input necessary to maintain the inside temperature at the target temperature
00781                 *supplementaryHeat= (targetTemperature - outsideAirTemp)*(GetspecificHeatCapAir()*airDensity * ventilationRate + meanThermalTrans*surfaceArea);

00788 void stable::uncontrolled(double heatOP,double outsideAirTemp, double windspeed, double solarRad)
00789 {
00790         double surfaceArea = planArea + wallArea; //sq meters - total surface area of walls and roof of housing
00791         double meanThermalTrans = (planArea * thermalTransRoof + wallArea * thermalTransWall)/surfaceArea;
00792 
00793         outsideAirTemp+=273.15;
00794         double skyTemp = outsideAirTemp; //apparent temperature of sky, K
00796         double surfaceTemp = outsideAirTemp; //rough estimate of temperature of roof surface, K
00797         double longWaveDown = GetlongWaveDown(skyTemp); //longwave radiation input to surface, Watts per square metre
00798         double longWaveUp = GetlongWaveUp(emissivity,surfaceTemp);//longwave radiation output of surface, Watts per square metre; note small error here, if surface temperature not equal to air temperature
00799         double longWave = longWaveDown - longWaveUp;//net longwave energy exchange, Watts per square metre
00800         double airDensity = GetdensityAir(GetstandardAirPressure(),outsideAirTemp,GetsaturatedWaterVapourPressure(outsideAirTemp));//density of the air in kg per cubic metre
00801         //should include a water balance!
00802         ventilationRate = 0; //cubic meters per sec
00803         double deltaTemper, tempDeltaTemp;
00806         double tempSol = outsideAirTemp + externSurfResis *(absorbCoeff *
00807                         solarRad - emissivity * longWave);  // eqn 9 in Cooper et al
00809         double deltaSol = tempSol - outsideAirTemp;
00811         double q = thermalTransRoof * planArea * deltaSol;//Watts
00813         double targetventilationRate=optimumAirVelocity* planArea;
00815         deltaTemper=(heatOP+q)/(GetspecificHeatCapAir()*airDensity* ventilationRate + meanThermalTrans*surfaceArea);
00816 
00818         if ((outsideAirTemp+deltaTemper>maxTemperature) && (outsideAirTemp<maxTemperature))
00819         {
00820                 targetventilationRate=(((heatOP+q)/(maxTemperature-outsideAirTemp))- meanThermalTrans*surfaceArea)
00821                                                                                                                 /(GetspecificHeatCapAir()*airDensity);
00822                 deltaTemper=(heatOP+q)/(GetspecificHeatCapAir()*airDensity* ventilationRate + meanThermalTrans*surfaceArea);
00823         }
00825         if (outsideAirTemp+deltaTemper<minTemperature)
00826         {
00827                 targetventilationRate=(((heatOP+q)/(minTemperature-outsideAirTemp))- meanThermalTrans*surfaceArea)
00828                                                                                                                 /(GetspecificHeatCapAir()*airDensity);
00829                 deltaTemper=(heatOP+q)/(GetspecificHeatCapAir()*airDensity* ventilationRate + meanThermalTrans*surfaceArea);
00830         }
00833         ventilationRate=targetventilationRate;
00834         double minVent;
00836         double apertureHeight = minPropApertureHeight * maxapertureHeight;
00837         CalcFreeVentilation(airDensity, meanThermalTrans, surfaceArea, apertureHeight,
00838                         windspeed, outsideAirTemp, heatOP, q, &minVent, &tempDeltaTemp);
00840         if (targetventilationRate<minVent)
00841         {
00842                 ventilationRate=minVent;
00843                 deltaTemper=tempDeltaTemp;
00844         }
00845         double maxVent;
00847         apertureHeight = maxapertureHeight;
00848         CalcFreeVentilation(airDensity, meanThermalTrans, surfaceArea, apertureHeight,
00849                         windspeed, outsideAirTemp, heatOP, q, &maxVent, &tempDeltaTemp);
00851         if (targetventilationRate>maxVent)

00862 void stable::CalcFreeVentilation(double airDensity, double thermalTrans, double surfaceArea, double apertureHeight,
00863                 double windspeed, double outsideAirTemp, double heatOP, double q, double *ventilationRate, double *deltaTemper)
00864 {
00866         *ventilationRate=0.0;
00867         double Cd = 0.6; 
00868 
00869         double B = 9.0 * outsideAirTemp/(pow(Cd,2)* pow(apertureWidth,2) * pow(apertureHeight,3) * 9.81);
00870 
00871         double  a1 = 0.0, b1 = 0.0, c1 = 0.0, d1 = 0.0;
00872         double  x[3];
00873         int     solutions;
00875         double spheatCap = GetspecificHeatCapAir();
00876         a1 =  spheatCap * airDensity;
00877         b1 = thermalTrans * surfaceArea;
00878 
00879 
00881         double windforcedVent = 0.25 * apertureHeight * apertureWidth * windspeed;  //we assume a square building, with only one wall facing the wind, cubic m per sec
00882         c1 = -pow(windforcedVent,2)* spheatCap * airDensity;
00883         d1 = -thermalTrans * surfaceArea * pow(windforcedVent,2) - (heatOP + q)/B;
00885         SolveCubic(a1, b1, c1, d1, &solutions, x);
00886         int index=0;
00887         double test=0.0;
00888         //solution is the highest value in the matrix
00889         for (int j=0; j<solutions; j++)
00890         {
00891                 if (x[j]>test)
00892                 {
00893                         test=x[j];
00894                         index=j;
00895                 }
00896 
00897                 *deltaTemper =  (heatOP + q)/(spheatCap * airDensity * x[index] + thermalTrans * surfaceArea);
00898                 *ventilationRate = x[index];
00899         }
00900 }
