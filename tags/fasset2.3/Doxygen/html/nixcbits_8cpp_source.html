00092 void Spline(double * x, double * y, int n, double * y2)
00093 {
00094    int i,k;
00095    double p, sig;
00096    double u[100];

00098    u[0] = 0.0;

00100    {
00101       sig = (x[i]-x[i-1])/(x[i+1]-x[i-1]);
00102       p = sig*y2[i-1]+2.0;
00103       y2[i] = (sig-1.0)/p;
00104       u[i] = (y[i+1]-y[i])/(x[i+1]-x[i])-(y[i]-y[i-1])/(x[i]-x[i-1]);
00105       u[i] = (6.0*u[i]/(x[i+1]-x[i-1])-sig*u[i-1])/p;
00106    }
00107    y2[n-1] = 0;
00108    for (k=n-2;k>=0;k--)
00109       y2[k] = y2[k]*y2[k+1]+u[k];
