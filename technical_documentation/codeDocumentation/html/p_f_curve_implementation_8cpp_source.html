00350 double pF_Curve_Implementation::CalcConductivity(double pFValue)
00351 {
00352  double pFLow = -1;
00353  double pFHigh = -1;
00354  double condLow = -1;
00355  double condHigh=0;
00356  pF_ConductionData * p;
00357  p = firstC;
00358  if (p)
00359  {
00360   condLow = condHigh = p->logConduc;
00361   pFLow = pFHigh = p->pF;
00362  };
00363  while (p)
00364  {
00365   if (pFHigh<pFValue)
00366   {
00367         condLow = condHigh;
00368         pFLow = pFHigh;
00369         condHigh = p->logConduc;
00370         pFHigh = p->pF;
00371   };
00372   p = p->next;
00373  };
00374  if (pFValue<pFLow)
00375   return condLow;     // set to watLow outside low boundary
00376  else if (pFValue>pFHigh)
00377   return 0;     // set to 0 outside high boundary
00378  else                                   // Linear interpolation
00379  {
00380   if (pFHigh==pFLow)
00381    return pow(10.,condLow);
00382   else
00383    return pow(10.,condLow+(condHigh-condLow)*(pFValue-pFLow)/(pFHigh-pFLow));
00384  }
00385 }
