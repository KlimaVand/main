00342 /****************************************************************************\
00343 Calculates the potential evaporation after a modified Pennman method.
00344 See: H.E. Mikkelsen & J.E. Olesen: Sammenligning af metoder til bestemmelse
00345 af potentiel fordampning. Statens Planteavlsfors�g. Beretning nr. S 2157. 1991
00346 globalRadiation = global radiation in MJ/m**2/day,
00347 temp = air temperature in Celsius at 2m,
00348 windSpeed = windspeed at 2m,
00349 humidity =  relative humidity as a proportion
00350 result = Evap in mm/day,
00351 \****************************************************************************/
00352 double climate::PenmanEvaporation(double globalRadiation,double temp,double windSpeed,double humidity)
00353 {
00354    double angle=DayAngle();
00355    const double evpHtWat=2.465;   //MJ/kg
00356    const double psykrConst=0.667;  //hPa/Celsius
00357    double B0=0.114-0.0659*cos(angle)+0.0451*sin(angle)-0.0851*cos(2*angle)-0.0031*sin(2*angle);
00358    double B1=0.083-0.4629*cos(angle)+0.0354*sin(angle)-0.0914*cos(2*angle)+0.0286*sin(2*angle);
00359    double B2=0.963-0.1707*cos(angle)+0.0123*sin(angle)-0.0419*cos(2*angle)+0.0381*sin(2*angle);
00360    double result =B0+B1*(DeltaVaporPress(temp)*globalRadiation/(evpHtWat*(DeltaVaporPress(temp)+psykrConst)))
00361                   +B2*(psykrConst*0.263*(0.5+0.54*windSpeed)*VaporPress(temp)*(1-humidity)
00362                       /(DeltaVaporPress(temp)+psykrConst));
00363    return result;
00364 }

00380 /****************************************************************************\
00381 Open water evaporation
00382 \****************************************************************************/
00383 double climate::Evaporation(double radiation,double temp,double humidity, double transferCoefficient)
00384 {
00385    //air density = (mole weight * atm. pressure)/(gas const. * Tair )
00386    double air_density = (28.96 * 101.3)/(8.314* (temp+273.0));          // kg per cubic m
00387    const double specific_heat= 1010.0;                                  // J/kg/K
00388    const double evpHtWat=2465000.0;                                     // J per kg
00389    const double psykrConst=66.7;                                        // Pa per K
00390    double Delta = DeltaVaporPress(temp) * 100.0;                        // mult by 100 to convert from mbars/K to Pa/K
00391    double vp = VaporPress(temp) * 100.0;                          // mult by 100 to convert from mbars to Pa
00392    double result =(Delta*radiation + air_density * specific_heat* vp * (1-humidity) *
00393                            transferCoefficient)/((Delta + psykrConst) * evpHtWat);
00394    return result;                                                                                                                               // result is in kg per square metre per second
00395 }
